# Level-manager-for-unity-
using System.Collections; using System.Collections.Generic; using UnityEngine; using UnityEngine.AddressableAssets;  public class Finish : MonoBehaviour {     private readonly List&lt;GameObject> levels = new List&lt;GameObject>();     int index;     [SerializeField] private GameObject player;     public static Finish Instance { get; private set; }      private void Start()     {         InstantiateLevel(index);         Dublicate();     }     private void Dublicate()     {         if (!Instance)         {             Instance = this;             DontDestroyOnLoad(this);         }         else         {             Destroy(gameObject);         }     }     private void OnCollisionEnter(Collision collision)     {         if (collision.gameObject.CompareTag("Player"))         {             index++;             InstantiateLevel(index);             DestroyLevel(index - 1);         }       }     public void InstantiateLevel(int index)     {         var position = new Vector3(0, 0);         Addressables.InstantiateAsync($"Level {index}", position, Quaternion.identity).Completed += handle =>         {             levels.Add(handle.Result);         };     }     private void DestroyLevel(int index)     {         Addressables.ReleaseInstance(levels[index]);     }
